#1. Hitunglah PageRank dari graph berikut !

import numpy as np

# Definisi matriks transisi
P = np.array([
    [0, 1/2, 1, 1/3,  0],
    [0, 0,   0, 1/3,  1/3],
    [0, 1/2, 0,   0,  1/3],
    [1, 0,   0,   0,  1/3],
    [0, 0,   0, 1/3,  0]
])
n = P.shape[0]

d = 0.85

# Inisialisasi PageRank
PR = np.ones(n) / n

for _ in range(100):
    PR = (1 - d) / n + d * P.dot(PR)

# Normalisasi
PR /= PR.sum()

# Urutan PageRank
ranked_nodes = np.argsort(-PR)

print("PageRank:")
for i, node in enumerate(ranked_nodes, start=1):
    print(f"Node {node + 1}: {PR[node]}")

#2. Carilah penyelesaian dari Sistem persamaan berikut dengan Eliminasi Gauss!

import numpy as np

# Membuat matriks augmented
matriks = np.array([
    [1, -1, 1, 3],
    [2, 1, 3, 8],
    [6, 0, -1, 4]], dtype=float)

# Fungsi untuk melakukan eliminasi Gauss
def eliminasi_gauss(matriks):
    n = len(matriks)
    for i in range(n):
        # Mencari pivot terbesar di kolom i
        baris_maks = np.argmax(np.abs(matriks[i:, i])) + i
        matriks[[i, baris_maks]] = matriks[[baris_maks, i]]

        # Membuat pivot menjadi 1
        matriks[i] = matriks[i] / matriks[i, i]

        # Mengeliminasi entri di bawah pivot
        for j in range(i+1, n):
            matriks[j] = matriks[j] - matriks[i] * matriks[j, i]

    # Substitusi balik
    solusi = np.zeros(n)
    for i in range(n-1, -1, -1):
        solusi[i] = matriks[i, -1] - np.sum(matriks[i, i+1:-1] * solusi[i+1:])

    return solusi

# Menyelesaikan sistem persamaan
solusi = eliminasi_gauss(matriks)

# Menghilangkan 0 di belakang koma
solusi_bulat = [int(x) if x.is_integer() else x for x in solusi]

print("Solusi sistem persamaan adalah:")
print(f"x1 = {solusi_bulat[0]}")
print(f"x2 = {solusi_bulat[1]}")
print(f"x3 = {solusi_bulat[2]}")
